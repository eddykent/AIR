--
--Query for getting 15 minute candles from the database from exchange_value_tick
--
WITH pre_data AS (
	SELECT the_date, open_price, high_price, low_price, close_price, from_currency || '/' || to_currency AS pair, EXTRACT(HOUR FROM the_date) AS h 
	FROM exchange_value_tick 
	WHERE the_date <= date '2020-05-12'::date + INTERVAL '15 hours' AND the_date >= date '2020-05-12'::date - INTERVAL '157.0 days' + INTERVAL '15 hours' 
	AND from_currency || '/' || to_currency = ANY(ARRAY['EUR/GBP','USD/JPY','AUD/CHF'])
),
maxs AS (
	SELECT pair, MAX(high_price) AS value FROM pre_data
	GROUP BY pair
),
mins AS (
	SELECT pair, MIN(low_price) AS value FROM pre_data
	GROUP BY pair
),
stepped AS (
	SELECT *,
	((LEAD(open_price) OVER(PARTITION BY pair ORDER BY the_date) - open_price) / open_price)*100 AS open_percent_change,
	((LEAD(high_price) OVER(PARTITION BY pair ORDER BY the_date) - high_price) / high_price)*100 AS high_percent_change,
	((LEAD(low_price) OVER(PARTITION BY pair ORDER BY the_date) - low_price) / low_price)*100 AS low_percent_change,
	((LEAD(close_price) OVER(PARTITION BY pair ORDER BY the_date) - close_price) / close_price)*100 AS close_percent_change,
	((high_price - open_price) / open_price)*100 AS high_local_percent_change,
	((low_price - open_price) / open_price)*100 AS low_local_percent_change
	FROM pre_data ORDER BY the_date DESC
)
SELECT the_date, 
JSON_OBJECT_AGG(
	stepped.pair, JSON_BUILD_OBJECT(
		'open_price', ROUND(open_price::NUMERIC,5),  
		'open_percent_change',ROUND(open_percent_change::NUMERIC,10),
		'high_price', ROUND(high_price::NUMERIC,5),  
		'high_percent_change',ROUND(high_percent_change::NUMERIC,10),
		'low_price', ROUND(low_price::NUMERIC,5),  
		'low_percent_change',ROUND(low_percent_change::NUMERIC,10),
		'close_price', ROUND(close_price::NUMERIC,5),  
		'close_percent_change',ROUND(close_percent_change::NUMERIC,10),
		'high_local_percent_change',ROUND(high_local_percent_change::NUMERIC,10),
		'low_local_percent_change',ROUND(low_local_percent_change::NUMERIC,10),
		'scaled_value',ROUND(((close_price - mins.value) / (maxs.value - mins.value))::NUMERIC,10),
		'instrument',stepped.pair
	)
	ORDER BY stepped.pair
) AS prices FROM stepped 
JOIN mins ON mins.pair = stepped.pair
JOIN maxs ON maxs.pair = stepped.pair
GROUP BY the_date ORDER BY the_date DESC;
